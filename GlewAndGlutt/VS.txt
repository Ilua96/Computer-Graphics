layout (location = 0) in vec3 position;
layout (location = 1) in vec3 normal;  
uniform mat4 mvp; 
uniform mat4 model;
out vec3 Normal;
out vec3 frag_pos;
void main(void)
{
	gl_Position = mvp * vec4(position, 1.0); 
	Normal = /*mat3(transpose(inverse(model))) *  */ normal; 
	frag_pos = vec3(model * vec4(position, 1.0f));
}
